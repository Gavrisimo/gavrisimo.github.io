Location on Windows: `C:/UsersUsername.gitconfig`

Location on Mac: `nano ~/.gitconfig`

Some useful shortcuts... :)

{% highlight bash linenos %}
[alias]
	co = checkout
	ci = commit
	st = status
	br = branch
	hist = log --pretty=format:&quot;%h %ad | %s%d [%an]&quot; --graph --date=short
	type = cat-file -t
	dump = cat-file -p
{% endhighlight %}
